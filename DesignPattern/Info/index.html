class EventEmitter {
  constructor() {
    this.cache = {}
  }

  on(name, fn) {
    const tasks = this.cache[name]
    if (tasks) {
      this.cache[name].push(fn)
    } else {
      this.cache[name] = [fn]
    }
  }

  off(name, fn) {
    const tasks = this.cache[name]
    if (tasks) {
      const index = tasks.findIndex(item => item === fn)
      if (index >= 0) {
        this.cache[name].splice(index, 1)
      }
    }
  }

  emit(name, ...args) {
    // å¤åˆ¶ä¸€ä»½ã€‚é˜²æ­¢å›è°ƒé‡Œç»§ç»­onï¼Œå¯¼è‡´æ­»å¾ªç¯

      }
    }
  }

  once(name, cb) {
    function fn(...args) {
      cb(args)
      this.off(name, fn)
    }
    this.on(name, fn)
  }
}
